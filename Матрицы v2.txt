#include <iostream>
using namespace std;

int main()
{

	setlocale(0, "");
	int i, j;
	int x, y, z;
	int a, **arr;
	int n; // размер массива 
	cout << "Введите размер массива: ";
	cin >> n; // получение от пользователя размера массива 

	arr = new int*[n]; // Выделение памяти для массива
	for (int i = 0; i < n; i++)
		arr[i] = new int[n];

	// Ввод данных массива 

	for (i = 0; i < n; i++)
	{
		for (j = 0; j < n; j++)
		{
			cout << "Введите число массива: ";
			cin >> arr[i][j];
		}
	}
	// Вывод массива 
	cout << " " << endl;

	for (i = 0; i < n; i++)
	{
		for (j = 0; j < n; j++)
		{
			cout << arr[i][j] << " ";
		}
		cout << endl;
	}

		// строка с мин элементом 
		x = arr[0][0];

		for (i = 0; i < n; i++)
		{
			for (j = 0; j < n; j++)
			{
				if (arr[i][j] <= x)
				{
					x = arr[i][j];
					y = i;
				}
			}
		}
		cout << "-----------------------------------------— " << endl;
		cout << "Строка с минимальным элементом y = " << y+1 << endl;
		cout << "Минимальный элемент x = " << x << endl;
		cout << "-----------------------------------------— " << endl;

		// столбец с макс элементом 

		x = arr[0][0];

		for (j = 0; j < n; j++)
		{
			for (i = 0; i < n; i++)
			{
				if (arr[i][j] >= x) {
					x = arr[i][j];
					z = i;
				}
			}
		}
		cout << "-----------------------------------------— " << endl;
		cout << "Столбец с максимальным элементом z = " << z+1 << endl;
		cout << "Максимальный элемент x = " << x << endl;
		cout << "-----------------------------------------— " << endl;

		// замена местами столбца и строки 
		int temp;
		for (i = 0; i < n; i++)
		{
			temp = arr[y][i];
			arr[y][i] = arr[i][z];
			arr[i][z] = temp;
		}
		// Вывод измененного массива 
		for (i = 0; i < n; i++)
		{
			for (j = 0; j < n; j++)
			{
				cout << arr[i][j] << " ";
			}
			cout << endl;
		}
		cout << "-----------------------------------------— " << endl;
		system("pause");
		return 0;
	
}
